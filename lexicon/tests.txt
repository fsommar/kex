## Run with python -m doctest tests.txt

>>> from lexicon import analyze
>>> analyze("It is a good day") > 0
True
>>> analyze("It is a bad day") < 0
True

## Test negation
>>> analyze("My overall opinion was positive") > 0
True
>>> analyze("My overall opinion was not positive") > 0
False
>>> analyze("Not bad!") > 0
True
>>> analyze("The performance was not very good") > 0
False

## Test spellcheck
>>> analyze("I havv gd spelingg") > 0

## Test intesifiers
>>> analyze("I strongly dislike winter") < analyze("I dislike winter")
>>> analyze("I really love sunshine") > analyze("I love sunshine")
